#include <Servo.h>
#include <Wire.h>
#include <LiquidCrystal_I2C.h>

// ---------------- RFID ----------------
#include <SPI.h>
#include <MFRC522.h>

#define SS_PIN A0
#define RST_PIN 255  // بدون پایه ریست
MFRC522 rfid(SS_PIN, RST_PIN);

String lastCard = "";
bool cardOK = false;
// -------------------------------------

LiquidCrystal_I2C lcd(0x27, 16, 2);
Servo myservo;

#define ir_enter 2
#define ir_back  4

#define ir_car1 5
#define ir_car2 6
#define ir_car3 7
#define ir_car4 8
#define ir_car5 9
#define ir_car6 A2   // ←← فقط همین یک تغییر!!!

int S1=0, S2=0, S3=0, S4=0, S5=0, S6=0;
int flag1=0, flag2=0;
int totalSlots = 6;
int slot = 6;
bool parkingFullShown = false;

void setup() {
  Serial.begin(9600);

  // --------- RFID ----------
  SPI.begin();
  rfid.PCD_Init();
  // -------------------------

  pinMode(ir_car1, INPUT);
  pinMode(ir_car2, INPUT);
  pinMode(ir_car3, INPUT);
  pinMode(ir_car4, INPUT);
  pinMode(ir_car5, INPUT);
  pinMode(ir_car6, INPUT);

  pinMode(ir_enter, INPUT);
  pinMode(ir_back, INPUT);

  myservo.attach(3);
  myservo.write(180);

  lcd.init();
  lcd.backlight();

  lcd.setCursor(0, 0);
  lcd.print("Car Parking");
  lcd.setCursor(1, 1);
  lcd.print("System");
  delay(2000);
  lcd.clear();

  displayscreen();
}

void loop() {
  Read_Sensor();

  int total = S1 + S2 + S3 + S4 + S5 + S6;
  slot = totalSlots - total;

  lcd.setCursor(0, 0);
  if (S1 == 1) lcd.print("S1:F "); else lcd.print("S1:E ");
  lcd.setCursor(5, 0);
  if (S2 == 1) lcd.print("S2:F "); else lcd.print("S2:E ");
  lcd.setCursor(10, 0);
  if (S3 == 1) lcd.print("S3:F "); else lcd.print("S3:E ");

  lcd.setCursor(0, 1);
  if (S4 == 1) lcd.print("S4:F "); else lcd.print("S4:E ");
  lcd.setCursor(5, 1);
  if (S5 == 1) lcd.print("S5:F "); else lcd.print("S5:E ");
  lcd.setCursor(10, 1);
  if (S6 == 1) lcd.print("S6:F "); else lcd.print("S6:E ");

  if (slot == 0 && !parkingFullShown) {
    lcd.clear();
    lcd.setCursor(0, 0);
    lcd.print("Parking Full!");
    parkingFullShown = true;
    delay(1500);
    lcd.clear();
  }

  if (slot > 0 && parkingFullShown) {
    parkingFullShown = false;
  }

  // ---------------- RFID READER ----------------
  if (rfid.PICC_IsNewCardPresent() && rfid.PICC_ReadCardSerial()) {
    lastCard = "";
    for (byte i = 0; i < rfid.uid.size; i++) {
      lastCard += String(rfid.uid.uidByte[i], HEX);
    }
    lastCard.toUpperCase();
    Serial.println("CARD -> " + lastCard);

    lcd.clear();
    lcd.setCursor(0,0);
    lcd.print("Card OK!");
    lcd.setCursor(0,1);
    lcd.print(lastCard);

    cardOK = true;
    rfid.PICC_HaltA();
    delay(300);
    lcd.clear();
  }
  // ------------------------------------------------

  // ************** ENTRY CONTROL **************
  if (cardOK && digitalRead(ir_enter) == 0 && flag1 == 0) {
    if (slot > 0) {
      flag1 = 1;
      myservo.write(90);
    }
  }

  // ************** EXIT CONTROL **************
  if (digitalRead(ir_back) == 0){ //&& flag2 == 0) {
    flag2 = 1;
    myservo.write(90);
      delay(3000);
    myservo.write(180);
  }

  if (flag1 == 1 && flag2 == 1) {

   // delay(1000);
   // myservo.write(180);

    cardOK = false;

    flag1 = 0;
    flag2 = 0;
  }

  delay(100);
}

void clearscreen() {
  lcd.clear();
}

void displayscreen() {
  lcd.clear();
  lcd.setCursor(0, 0);
  lcd.print("*ENTER THE CODE*");
  lcd.setCursor(0, 1);
  lcd.print("TO OPEN DOOR!!");
}

void Read_Sensor() {
  S1 = (digitalRead(ir_car1) == 0);
  S2 = (digitalRead(ir_car2) == 0);
  S3 = (digitalRead(ir_car3) == 0);
  S4 = (digitalRead(ir_car4) == 0);
  S5 = (digitalRead(ir_car5) == 0);
  S6 = (digitalRead(ir_car6) == 0);   // ← همون تابع قبلی
}
